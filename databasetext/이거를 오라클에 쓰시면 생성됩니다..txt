CREATE TABLE BOARD(
    seq INT PRIMARY KEY,
    title VARCHAR(255) not null,
    content VARCHAR(1000) not null,
    writer VARCHAR(10) not null,
    password INT not null,
    regDate TIMESTAMP not null,
    cnt INT not null
    
)

CREATE SEQUENCE BOARD_SEQ
START WITH 1
INCREMENT BY 1
MINVALUE 1
MAXVALUE 99999;
